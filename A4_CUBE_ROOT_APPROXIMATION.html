/**

float absolute_value(float x)
{
    if (x < 0) return -1 * x;
    return x;

/**
 * Return the absolute value of (n - (b *  b * b)).
 */
long double difference(long double n, long double b)
{
    if (n > (b * b * b)) return (n - (b * b * b));
    return ((b * b * b) - n);
}
